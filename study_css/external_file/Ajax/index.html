Ajaxで読み取り

        
        <script>
                $(function() {
                    //html読み込み
                    $('#button').on('click', function() {
                        $.ajax({
                            url: 'sample2.html',
                            dataType: 'html',
                            success: function(data) {
                                $('#text').html(data);
                            },
                            error: function(data) {
                                alert('エラー');
                            }
                        });
                    });
                    //json読み込み
                    $('#button2').on('click', function() {
                        $.ajax({
                            url: 'sample3.json',
                            dataType: 'json',
                            success: function(data) {
                                if (!$('.chara_list').length) {
                                    for (var i in data) {
                                        var ele = data[i];
        
                                        if (ele.name == undefined) ele.name = '【不明】';
                                        if (ele.age == undefined) ele.age = '【不明】';
                                        if (ele.tel == undefined) ele.tel = '【不明】';
                                        if (ele.type == undefined) ele.type = '【不明】';
                                        var name = '<li class="name">名前：' + ele.name + '</li>';
                                        var age = '<li class="age">年齢：' + ele.age + '</li>';
                                        var tel = '<li class="tel">電話番号：' + ele.tel + '</li>';
                                        var type = '<li class="type">属性：' + ele.type + '</li>';
                                        var data_list = '<ul class="chara_list">' + name + age + tel + type + '</ul>';
                                        $('#chara').append(data_list);
                                    }
                                }
                            },
                            error: function(data) {
                                alert('エラー');
                            }
                        });
                    });
                });
            </script>
